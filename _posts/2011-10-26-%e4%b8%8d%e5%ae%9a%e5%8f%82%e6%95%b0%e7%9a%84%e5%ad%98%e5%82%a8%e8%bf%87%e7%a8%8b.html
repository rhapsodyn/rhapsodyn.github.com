CREATE PROCEDURE PROC_BLAHBLAH
@PRAM1 INT = NULL,
@PRAM2 INT = NULL
AS
BEGIN
SELECT * FROM TBL_BLAHBLAH
WHERE 1=1
   AND (COLUMN1 = @PRAM1 OR @PRAM1 IS NULL)
   AND (COLUMN2 = @PRAM2 OR @PRAM2 IS NULL)
END
